445  public Brush DrawHealthBar(double health, double wid, double hei)

537  public Brush DrawText(SolidColorBrush bg, string text, int fontSize, Rect panel, SolidColorBrush fontColor, Typeface fontTypeface, Pen panelBorder = null, Pen fontBorder = null, double opacity = 1)
538  {
539  var dg = new DrawingGroup();
540  int chars = text.Length;
541  double yalign = (panel.Height / 2) - (fontSize / 2);
542  double xalign = (panel.Width / 2) - (chars * fontSize * 0.6 * 0.5);
543  dg.Children.Add(new GeometryDrawing(bg, panelBorder, new RectangleGeometry(panel)));
544  dg.Children.Add(new GeometryDrawing(fontColor, fontBorder, new FormattedText(
545  text,
546  CultureInfo.CurrentCulture,
547  FlowDirection.LeftToRight,
548  fontTypeface,
549  fontSize,
550  fontColor,
551  1.25).BuildGeometry(new Point(xalign, yalign))));
552  DrawingBrush textBrush = new DrawingBrush();
553  textBrush.Drawing = dg;
554  textBrush.Opacity = opacity;
555  return textBrush;
556  }

565  public Brush GenerateName(Guid identifier, string name, int fontSize)
566  {

569  var textBrush = this.SimpleText(name, fontSize, Brushes.DarkRed, this.OfficialFont, this.ThinBlackBorder);

572  }
573 
574  return this.playerNames.Where(x => x.Key == identifier).First().Value;
575  }
