89  public void Update(double deltaTime)
90  {
91  var bullets = this.model.Bullets;
92  for (int i = 0; i < bullets.Count(); i++)
93  {
94  var bullet = bullets.ElementAt(i);
95  Vector start = bullet.StartPoint;
96  double x = bullet.Bounds.X;
97  double y = bullet.Bounds.Y;
98  double a = x - start.X;
99  double b = y - start.Y;
100  double suppression = 1 - (Math.Sqrt((a * a) + (b * b)) * bullet.Weight);
101  double movement = bullet.Speed * deltaTime * suppression;
102  Vector direction = bullet.Direction;
103  bullet.Bounds = new Rect(
104  x + (direction.X * movement),
105  y + (direction.Y * movement),
106  width: bullet.Bounds.Width,
107  bullet.Bounds.Height);
108  var items = this.model.Collision(bullet.Id);
109  if (items.Count() > 0)
110  {
111  this.model.DestructItem(bullet.Id);
112  var players = items.Where(z => z.Type == "Player");
113  if (players.Count() > 0)
114  {
115  this.gameServer.ReplyHandler(new Request() { Broadcast = true, Operation = Operation.Hit, Parameters = $"{players.First().Id};{bullet.Id}" });
116  }
117 
118  this.gameServer.ReplyHandler(new Request() { Broadcast = true, Operation = Operation.Remove, Parameters = $"{bullet.Id}" });
119  }
120  }
