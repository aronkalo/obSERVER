240  public void FlyBullets(double deltaTime)
241  {
242  this.model.LoadCache();
243  var bullets = this.model.Bullets;
244  int c = bullets.Count();
245  for (int i = 0; i < bullets.Count(); i++)
246  {
247  var ibullet = (IBullet)bullets.ElementAt(i);
248  if (ibullet.RealSpeed < 80)
249  {
250  this.model.DestructItem(ibullet.Id);
251  this.Render?.Invoke(this, null);
252  continue;
253  }
254 
255  var bounds = ibullet.RealPrimitive.Bounds;
256  ibullet.Fly(deltaTime);
257  var items = this.model.Intersection(bounds);
258  for (int y = 0; y < items.Count(); y++)
259  {
260  if (items.ElementAt(y).CollidesWith(bullets.ElementAt(i).RealPrimitive))
261  {
262  if (items.ElementAt(y).GetType() == typeof(Player))
263  {
264  this.gameClient.Send(Operation.Hit, $"{items.ElementAt(y).Id};{ibullet.Id};{ibullet.BulletDamage};{this.Milis}");
265  this.model.DestructItem(ibullet.Id);
266  break;
267  }
268  else
269  {
270  this.gameClient.Send(Operation.Remove, $"{ibullet.Id}");
271  this.model.DestructItem(ibullet.Id);
272  break;
273  }
274  }
