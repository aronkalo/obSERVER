297  public void LoadItems(string fileName)
298  {
299  XDocument xDoc = XDocument.Load(fileName);
300  foreach (var node in xDoc.Root.Descendants("item"))
301  {
302  string type = node.Attribute("type").Value;
303  double x = double.Parse(node.Element("x").Value.Replace('.', ','));
304  double y = double.Parse(node.Element("y").Value.Replace('.', ','));
305  double width = double.Parse(node.Element("width").Value.Replace('.', ','));
306  double height = double.Parse(node.Element("height").Value.Replace('.', ','));
307  double angle = double.Parse(node.Element("angle").Value);
308  bool impact = bool.Parse(node.Element("impact").Value);
309  Guid id = Guid.Parse(node.Attribute("id").Value);
310  if (type == "crate")
311  {
312  XElement xWeapon = node.Element("weapon");
313  Guid wId = Guid.Parse(xWeapon.Attribute("id").Value);
314  int capacity = int.Parse(xWeapon.Attribute("capacity").Value.Replace('.', ','));
315  double speed = double.Parse(xWeapon.Attribute("speed").Value.Replace('.', ','));
316  int shoot = int.Parse(xWeapon.Attribute("shoot").Value.Replace('.', ','));
317  double damage = double.Parse(xWeapon.Attribute("damage").Value.Replace('.', ','));
318  double weight = double.Parse(xWeapon.Attribute("weight").Value.Replace('.', ','));
319  IWeapon weapon = new Weapon(new EllipseGeometry() { RadiusX = 1, RadiusY = 1 }, wId, new double[] { x + (width / 2), y + (height / 2) }, 0, false, capacity, shoot, speed, damage, weight);
320  weapon.Owned = true;
321  ICrate crate = new Crate(new RectangleGeometry(new Rect(0, 0, width, height)), id, new double[] { x, y }, angle, impact);
322  crate.ContainedWeapon = weapon;
323  this.ConstructItem(weapon, weapon.GetType());
324  this.ConstructItem(crate, crate.GetType());
325  }
326  else
327  {
328  this.ConstructItem(new StaticItem(new RectangleGeometry() { Rect = new System.Windows.Rect(0, 0, width, height) }, id, new double[] { x, y }, angle, new double[] { width, height }, impact, type), typeof(StaticItem));
329  }
330  }
331  }
