376  internal void DrawElements(DrawingContext context)
377  {
378  DrawingGroup dGroup = new DrawingGroup();
379 
380  dGroup.Transform = new TranslateTransform(-(this.model.MyPlayer.Position.X - this.xCenter), -(this.model.MyPlayer.Position.Y - this.yCenter));
381 
382  dGroup.Children.Add(this.drawingCache.MapInstance);
383  dGroup.Children.Add(this.drawingCache.GroundInstance);
384  dGroup.Children.Add(this.drawingCache.MiddleInstance);
385 
386  foreach (var weapon in this.model.Weapons)
387  {
388  dGroup.Children.Add(this.WeaponDrawing((IWeapon)weapon));
389  }
390 
391  foreach (var Bullet in this.model.Bullets)
392  {
393  GeometryDrawing bullet = new GeometryDrawing(this.textureCache.BulletBrush, this.textureCache.BlackBorder, Bullet.RealPrimitive);
394  dGroup.Children.Add(bullet);
395  }
396 
397  foreach (var player in this.model.Players)
398  {
399  var Player = player as IPlayer;
400  ImageBrush texture = this.textureCache.PlayerBrush.Clone();
401  if (Player.CurrentWeapon != null)
402  {
403  switch (Player.CurrentWeapon.WeaponInfo[0])
404  {
405  case 8:
406  texture = this.textureCache.PlayerShotgunBrush.Clone();
407  break;
408  case 20:
409  texture = this.textureCache.PlayerPistolBrush.Clone();
410  break;
411  case 30:
412  texture = this.textureCache.PlayerRifleBrush.Clone();
413  break;
414  }
415  }
416 
417  Geometry primitive = Player.RealPrimitive;
418  texture.Transform = primitive.Transform;
419  GeometryDrawing playerPrimitive = new GeometryDrawing(texture, null, primitive);
420  dGroup.Children.Add(playerPrimitive);
421  dGroup.Children.Add(this.textureCache.DrawTextUp(primitive.Bounds, this.textureCache.GenerateName(Player.Id, Player.Name, 18), Player.Name, 18));
422  }
423 
424  dGroup.Children.Add(this.drawingCache.TopInstance);
425  dGroup.Children.Add(this.DrawWeaponStat());
426  var clouds = this.drawingCache.AtmosphereInstance;
427  clouds.Geometry.Transform = new TranslateTransform(-(this.model.MyPlayer.Position.X - this.xCenter) / 5, -(this.model.MyPlayer.Position.Y - this.yCenter) / 5);
428  dGroup.Children.Add(clouds);
429  dGroup.Children.Add(this.DrawHealthBar());
430 
431  context.DrawDrawing(dGroup);
432  }

434  private GeometryDrawing DrawWeaponStat()
435  {
436  DrawingGroup dGroup = new DrawingGroup();
437  Vector pos = this.model.MyPlayer.Position;
438  double pWidth = this.xCenter * 0.25;
439  double pHeight = this.yCenter * 0.4;
440  Rect panel = new Rect(pos.X + (this.xCenter * 0.75) - 20, pos.Y + (this.yCenter * 0.6) - 20, pWidth, pHeight);
441  Rect weaponBulletPanel = new Rect(panel.X + (panel.Width * 0.02), panel.Y + (panel.Height * 0.5), panel.Width * 0.96, panel.Height * 0.2);
442  Rect storedBulletPanel = new Rect(panel.X + (panel.Width * 0.02), panel.Y + (panel.Height * 0.8), panel.Width * 0.96, panel.Height * 0.1);
443  dGroup.Children.Add(new GeometryDrawing(Brushes.DarkGreen, this.textureCache.BlackBorder, new RectangleGeometry(panel)));
444  dGroup.Children.Add(new GeometryDrawing(Brushes.Brown, this.textureCache.BlackBorder, new RectangleGeometry(weaponBulletPanel)));
445  dGroup.Children.Add(new GeometryDrawing(Brushes.DarkRed, this.textureCache.BlackBorder, new RectangleGeometry(storedBulletPanel)));
446 
447  if (this.model.MyPlayer.CurrentWeapon != null)
448  {
449  double[] parameters = this.model.MyPlayer.CurrentWeapon.WeaponInfo;
450  switch (parameters[0])
451  {
452  case 8:
453  dGroup.Children.Add(new GeometryDrawing(this.textureCache.ShotgunSideBrush, null, new RectangleGeometry(new Rect(panel.X, panel.Y, panel.Width, panel.Height * 0.5))));
454  break;
455  case 20:
456  dGroup.Children.Add(new GeometryDrawing(this.textureCache.PistolSideBrush, null, new RectangleGeometry(new Rect(panel.X, panel.Y, panel.Width, panel.Height * 0.5))));
457  break;
458  case 30:
459  dGroup.Children.Add(new GeometryDrawing(this.textureCache.RifleSideBrush, null, new RectangleGeometry(new Rect(panel.X, panel.Y, panel.Width, panel.Height * 0.5))));
460  break;
461  }
462 
463  dGroup.Children.Add(new GeometryDrawing(this.textureCache.DrawWeaponBullets((int)parameters[5], (int)parameters[0], "Weapon "), null, new RectangleGeometry(weaponBulletPanel)));
464  }
465 
466  dGroup.Children.Add(new GeometryDrawing(this.textureCache.DrawBullets(this.model.MyPlayer.StoredBullets, "Stored Bullets: "), null, new RectangleGeometry(storedBulletPanel)));
467  DrawingBrush dBrush = new DrawingBrush();
468  dBrush.Drawing = dGroup;
469  dBrush.Opacity = 0.7;
470  return new GeometryDrawing(dBrush, null, new RectangleGeometry(panel));
471  }
